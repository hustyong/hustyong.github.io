func ValidateHeader(header string) gin.HandlerFunc {
    return func(c *gin.Context) {
        if value := c.GetHeader(header); value == "" {
            c.AbortWithStatusJSON(http.StatusBadRequest, gin.H{
                "message": fmt.Sprintf("missing %s header", header),
            })
        } else {
            c.Next()
        }
    }
}
